{\tt dvipdfmx} での処理では，{\tt pgfplot}で以下のような等高線を表示できなかった．
\begin{figure}[h]
\begin{tikzpicture}
  \begin{axis}[
    title={$x \exp(-x^2-y^2)$ and its gradient},
    domain=-2:2,
    view={0}{90},
    axis background/.style={fill=white},
    ]
    \addplot3[contour gnuplot={number=9,
      labels=false},thick] 
    {exp(0-x^2-y^2)*x};
    \addplot3[blue,
    quiver={
      u={exp(0-x^2-y^2)*(1-2*x^2)},
      v={exp(0-x^2-y^2)*(-2*x*y)},
      scale arrows=0.3,
    },
    -stealth,samples=15]
    {exp(0-x^2-y^2)*x};
    \node [fill=white] at (axis cs:0,0) {日本語で注釈};
  \end{axis}
\end{tikzpicture}
\end{figure}
